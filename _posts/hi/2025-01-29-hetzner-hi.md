एक सर्वर इनस्टांस डिटेल्स प्राप्त करने के लिए एक सरल पायथन स्क्रिप्ट।

```python
from hcloud import Client
import os

# एनवायरन्मेंट वैरिएबल से API टोकन प्राप्त करें
api_token = os.environ.get('HERTZNER_API_KEY')

यदि नहीं api_token:
    अनुचित करें: "HERTZNER_API_KEY एनवायरन्मेंट वैरिएबल नहीं सेट है।"
    कार्यान्वयन 1

# क्लाइंट इनस्टांस बनाएं
client = Client(token=api_token)

# सभी सर्वर लिस्ट करें
servers = client.servers.get_all()

# सर्वर विवरण प्रिंट करें
सर्वर के लिए वर्तमान सर्वर में लूप करें:
    प्रिंट(f"सर्वर आईडी: {सर्वर.id}")
    प्रिंट(f"सर्वर नाम: {सर्वर.नाम}")
    प्रिंट(f"सर्वर स्टेटस: {सर्वर.स्टेटस}")
    प्रिंट(f"सर्वर IPv4: {सर्वर.public_net.ipv4.ip}")
    प्रिंट(f"सर्वर IPv6: {सर्वर.public_net.ipv6.ip}")
    प्रिंट(f"सर्वर प्रकार: {सर्वर.server_type.name}")
    प्रिंट(f"सर्वर स्थान: {सर्वर.datacenter.location.name}")
    प्रिंट("----------------------------------")

# अगर आपको एक विशेष सर्वर ID द्वारा सर्वर प्राप्त करना है
सर्वर आईडी = '59402674'
सर्वर = client.servers.get_by_id(सर्वर आईडी)

प्रिंट(f"विशेष सर्वर आईडी: {सर्वर.id}")
प्रिंट(f"विशेष सर्वर नाम: {सर्वर.नाम}")
प्रिंट(f"विशेष सर्वर स्टेटस: {सर्वर.स्टेटस}")
प्रिंट(f"विशेष सर्वर IPv4: {सर्वर.public_net.ipv4.ip}")
प्रिंट(f"विशेष सर्वर IPv6: {सर्वर.public_net.ipv6.ip}")
प्रिंट(f"विशेष सर्वर प्रकार: {सर्वर.server_type.name}")
प्रिंट(f"विशेष सर्वर स्थान: {सर्वर.datacenter.location.name}")

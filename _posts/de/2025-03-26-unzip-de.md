Entpacken von Dateien in Python:

```python
import zipfile
import os
import argparse

# Argumentparsing einrichten
parser = argparse.ArgumentParser(description="Entpacke eine Datei in dasselbe Verzeichnis mit demselben Namen.")
parser.add_argument('zip_file', help="Pfad zur Zip-Datei")

# Kommandozeilenargumente parsen
args = parser.parse_args()

# Pfad zur Zip-Datei und ihr Verzeichnis erhalten
zip_file_path = args.zip_file
directory = os.path.dirname(zip_file_path)

# .zip-Erweiterung entfernen, um den neuen Verzeichnis/Ordner-Namen zu erhalten
folder_name = os.path.splitext(os.path.basename(zip_file_path))[0]
destination_folder = os.path.join(directory, folder_name)

# Sicherstellen, dass der Zielordner existiert
os.makedirs(destination_folder, exist_ok=True)

# Datei entpacken
with zipfile.ZipFile(zip_file_path, 'r') as zip_ref:
    zip_ref.extractall(destination_folder)

print(f'Inhalt extrahiert nach {destination_folder}')

```

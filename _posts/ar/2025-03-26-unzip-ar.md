فك ضغط الملفات في Python:

```python
import zipfile
import os
import argparse

# إعداد تحليل الوسائط
parser = argparse.ArgumentParser(description="Unzip a file to the same directory with the same name.")
parser.add_argument('zip_file', help="Path to the zip file")

# تحليل الوسائط من سطر الأوامر
args = parser.parse_args()

# الحصول على مسار ملف الضغط والمجلد الذي يحتويه
zip_file_path = args.zip_file
directory = os.path.dirname(zip_file_path)

# إزالة الامتداد .zip للحصول على اسم المجلد/الدليل الجديد
folder_name = os.path.splitext(os.path.basename(zip_file_path))[0]
destination_folder = os.path.join(directory, folder_name)

# التأكد من وجود مجلد الوجهة
os.makedirs(destination_folder, exist_ok=True)

# فك ضغط الملف
with zipfile.ZipFile(zip_file_path, 'r') as zip_ref:
    zip_ref.extractall(destination_folder)

print(f'Contents extracted to {destination_folder}')

```

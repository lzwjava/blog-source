Descomprimir archivos en Python:

```python
import zipfile
import os
import argparse

# Configurar el análisis de argumentos
parser = argparse.ArgumentParser(description="Descomprime un archivo en el mismo directorio con el mismo nombre.")
parser.add_argument('zip_file', help="Ruta al archivo zip")

# Analizar los argumentos de la línea de comandos
args = parser.parse_args()

# Obtener la ruta del archivo zip y su directorio
zip_file_path = args.zip_file
directory = os.path.dirname(zip_file_path)

# Eliminar la extensión .zip para obtener el nuevo nombre del directorio/carpeta
folder_name = os.path.splitext(os.path.basename(zip_file_path))[0]
destination_folder = os.path.join(directory, folder_name)

# Asegurar que la carpeta de destino existe
os.makedirs(destination_folder, exist_ok=True)

# Descomprimir el archivo
with zipfile.ZipFile(zip_file_path, 'r') as zip_ref:
    zip_ref.extractall(destination_folder)

print(f'Contenido extraído a {destination_folder}')

```
